/** * DataPulse Suite Pro - Enterprise Backend Engine * Arquitectura unificada para control estratégico, inventarios, OKRs y ventas. */ function doPost(e) { // Manejo de CORS preflight manual si fuera necesario var output = ContentService.createTextOutput(); output.setMimeType(ContentService.MimeType.JSON); try { var params = JSON.parse(e.postData.contents); var action = params.action; var userEmail = params.correo; var userId = params.idUsuario; if (!userEmail || !userId) { return responseError("Error de Autenticación: Credenciales vacías."); } // Obtener o crear de forma automática la hoja de cálculo para este usuario var ss = obtenerBaseDatos(userEmail, userId); switch(action) { case 'login': return responseSuccess(inicializarTodo(ss)); case 'guardarOnboarding': case 'guardarPerfil': actualizarDatosPerfil(ss, params.organizacion); return responseSuccess(inicializarTodo(ss)); case 'registrarVenta': insertarFila(ss, "Ventas", [ params.venta.fecha, params.venta.producto, params.venta.vendedor, params.venta.ubicacion, parseFloat(params.venta.monto), parseInt(params.venta.cantidad) ]); return responseSuccess(inicializarTodo(ss)); case 'registrarProducto': insertarFila(ss, "Productos", [ params.producto.nombre, parseInt(params.producto.stockInicial), parseFloat(params.producto.precio) ]); return responseSuccess(inicializarTodo(ss)); case 'registrarVendedor': insertarFila(ss, "Vendedores", [ params.vendedor.nombre, params.vendedor.zona ]); return responseSuccess(inicializarTodo(ss)); case 'registrarOKR': insertarFila(ss, "OKRs", [ params.okr.objetivo, params.okr.desc, parseFloat(params.okr.meta) ]); return responseSuccess(inicializarTodo(ss)); default: return responseError("Operación ejecutable no identificada."); } } catch(error) { return responseError("Excepción Crítica en Backend: " + error.toString()); } } /** * Busca una hoja existente en el Drive del administrador o crea una nueva con la estructura Base. */ function obtenerBaseDatos(email, id) { var nombreArchivo = "DataPulse_DB_" + id + "_" + email.replace(/[@.]/g, "_"); var archivos = DriveApp.getFilesByName(nombreArchivo); if (archivos.hasNext()) { return SpreadsheetApp.open(archivos.next()); } else { var nuevoSs = SpreadsheetApp.create(nombreArchivo); prepararEstructuraHojas(nuevoSs); return nuevoSs; } } /** * Inicializa las tablas estructurales si la base de datos es nueva. */ function prepararEstructuraHojas(ss) { var estructuras = { "Ventas": [["Fecha", "Producto", "Vendedor", "Ubicación", "Monto", "Cantidad"]], "Productos": [["Nombre", "Stock Inicial", "Precio"]], "Vendedores": [["Nombre", "Zona Asignada"]], "OKRs": [["Objetivo General", "Key Result", "Meta"]], "Organizacion": [["Propiedad", "Valor"]] }; // Modificar la primera hoja por defecto var hojasExistentes = ss.getSheets(); if(hojasExistentes.length > 0) { hojasExistentes[0].setName("Ventas"); hojasExistentes[0].getRange(1, 1, 1, estructuras["Ventas"][0].length).setValues(estructuras["Ventas"]); } for (var nombreHoja in estructuras) { if (!ss.getSheetByName(nombreHoja)) { var hoja = ss.insertSheet(nombreHoja); hoja.getRange(1, 1, 1, estructuras[nombreHoja][0].length).setValues(estructuras[nombreHoja]); } } } /** * Lee todo el ecosistema de datos del usuario y lo retorna estructurado para el Frontend. */ function inicializarTodo(ss) { return { ventas: obtenerDatosHoja(ss, "Ventas"), productos: obtenerDatosHoja(ss, "Productos"), vendedores: obtenerDatosHoja(ss, "Vendedores"), objetivos: obtenerDatosHoja(ss, "OKRs"), organizacion: obtenerDiccionarioPerfil(ss) }; } function obtenerDatosHoja(ss, nombreHoja) { var hoja = ss.getSheetByName(nombreHoja); if (!hoja) return []; var rango = hoja.getDataRange(); var valores = rango.getValues(); if (valores.length <= 1) return []; var cabeceras = valores[0]; var resultado = []; for (var i = 1; i < valores.length; i++) { var fila = {}; for (var j = 0; j < cabeceras.length; j++) { // Formatear fechas nativas a String ISO para que no rompa el JSON if (valores[i][j] instanceof Date) { fila[cabeceras[j]] = valores[i][j].toISOString().split('T')[0]; } else { fila[cabeceras[j]] = valores[i][j]; } } resultado.push(fila); } return resultado; } function obtenerDiccionarioPerfil(ss) { var hoja = ss.getSheetByName("Organizacion"); if (!hoja) return {}; var valores = hoja.getDataRange().getValues(); var datos = {}; for(var i = 1; i < valores.length; i++) { if(valores[i][0]) datos[valores[i][0]] = valores[i][1]; } return datos; } function actualizarDatosPerfil(ss, objetoPerfil) { var hoja = ss.getSheetByName("Organizacion"); hoja.clearContents(); hoja.getRange(1,1,1,2).setValues([["Propiedad", "Valor"]]); var filasAInsertar = []; for(var llave in objetoPerfil) { filasAInsertar.push([llave, objetoPerfil[llave]]); } if(filasAInsertar.length > 0) { hoja.getRange(2, 1, filasAInsertar.length, 2).setValues(filasAInsertar); } } function insertarFila(ss, nombreHoja, arrayValores) { var hoja = ss.getSheetByName(nombreHoja); if (!hoja) throw new Error("La tabla " + nombreHoja + " no existe."); hoja.appendRow(arrayValores); } function responseSuccess(data) { return ContentService.createTextOutput(JSON.stringify({ status: "success", data: data })) .setMimeType(ContentService.MimeType.JSON); } function responseError(msg) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: msg })) .setMimeType(ContentService.MimeType.JSON); }
